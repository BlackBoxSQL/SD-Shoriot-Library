# কিভাবে একটি নতুন প্রোজেক্ট শুরু করবেন?  

>>> cd Desktop

>>> mkdir Boss

>> cd Boss

>>> virtualenv venv --python=python3.7

>>> source venv/bin/activate

>>> pip install django==2.0.7

django-admin startproject project
cd project
python manage.py runserver
